Selected: {{(selected) ? selected : 'no'}} 

 <--hierarchical-select
         :nodes="nodes"
         :selected="selected"
         :change-select="changeSelect"
         :css-classes="['uk-select', 'uk-margin-bottom']"
 -->

new Vue({
    el: '#app',
    methods: {
      changeSelect: function (value) {
        this.selected = value;
      }
    },
    data: {
      selected: 5,
      nodes: [
        {
          name: 'test1',
          id: 1,
          children: [
            {name: 'test2', id: 2},
            {
              name: 'test3',
              id: 3,
              children: [
                {name: 'test4', id: 4},
                {name: 'test5', id: 5},
              ]
            },
          ]
        }
      ],
    }
})
